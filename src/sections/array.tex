    \section{Usuwanie duplikatów z tablicy posortowanej}
    \begin{minted}[mathescape,
               linenos,
               numbersep=5pt,
               gobble=2,
               frame=lines,
               % fontsize=\footnotesize,
               framesep=2mm]{java}
    void removeDuplicates(int[] arr, int n) {
        if (n == 0 || n == 1)
            return;

        int j = 0;

        for (int i = 0; i < n - 1; i++)
            if (arr[i] != arr[i + 1])
                arr[j++] = arr[i];

        arr[j++] = arr[n - 1];
        n = j;
    }
   \end{minted}
